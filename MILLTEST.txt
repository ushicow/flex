*
* MILL RESET TEST
*
P.KEY	.EQ $8006	FOR KEY INPUT
P.OUT	.EQ $8007	FOR DISPLAY OUT
P.DOS	.EQ $8008	FOR APPLE DOS
P.FLAG	.EQ $8009	FOR MONITOR

	.OR $0000
	.TA $8000
	.TF MILLTEST.8000

	LDX #$C000
I1	CLR ,X+
	INC $8400
	CMPX #$0000
	BNE I1
I	LDX #$C000
I2	LDA ,X+
	BNE ERROR
	CMPX #$0000
	BEQ I
	INC $8401
	BRA I2
ERROR	STX $0000
	INC P.FLAG
I3	EXG D,D
	BRA I3
	.EN
